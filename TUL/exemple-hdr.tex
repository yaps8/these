\documentclass{thesul}
\begin{document}
\HdrUL
\ThesisTitle{Les travaux que j'ai dirigés}

\Rapporteurs  = {Le rapporteur 1\\
                 Le rapporteur 2\\
                 Le rapporteur 3}
\Examinateurs = {L'examinateur 1\\
                 L'examinateur 2}

\MakeThesisTitlePage
\end{document}
